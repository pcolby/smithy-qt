21bool Renderer::loadTemplates(const QString &dir)
22{
23 qCInfo(lc).noquote() << tr("Loading Grantlee templates from %1").arg(dir);
24
25 // Setup the template loader.
26 auto loader = QSharedPointer<Grantlee::FileSystemTemplateLoader>::create();
27 // Note, {% include "<filename>" %} will look for files relative to templateDirs.
28 loader->setTemplateDirs(QStringList{dir});
29 auto cachedLoader = QSharedPointer<Grantlee::CachingLoaderDecorator>::create(loader);
30 engine.addTemplateLoader(cachedLoader);
31
32 // Load the templates.
33 QDirIterator iter{QDir::cleanPath(dir), QDir::Files, QDirIterator::Subdirectories};
34 while (iter.hasNext()) {
35 const QString name = iter.next().mid(iter.path().size()+1);
36 qCDebug(lc).noquote() << tr("Loading template: %1").arg(name);
37 const Grantlee::Template tmplate = engine.loadByName(name);
38 if (tmplate->error()) {
39 qCritical().noquote() << tr("Error loading template %1: %2")
40 .arg(name, tmplate->errorString());
41 return false;
42 }
43 templates.append(name);
44 }
45 qDebug(lc).noquote() << tr("Loaded %n template/s from %1", nullptr, templates.size()).arg(dir);
46 return true;
47}

81bool Renderer::render(const QString &templateName, const QString &outputPathName,
82 const QVariantHash &additionalContext)
83{
84 qCDebug(lc).noquote() << tr("Rendering %1 to %2").arg(templateName, outputPathName);
85 if (!templates.contains(templateName)) {
86 qCCritical(lc).noquote() << tr("Template %1 has not been loaded").arg(templateName);
87 return false;
88 }
89
90 const QDir dir = QFileInfo(outputPathName).dir();
91 if (!dir.mkpath(QSL("./"))) {
92 qCCritical(lc).noquote() << tr("Failed to create directory path %1").arg(dir.path());
93 return false;
94 }
95
96 QFile file(outputPathName);
97 if (!file.open(QFile::WriteOnly)) {
98 qCCritical(lc).noquote() << tr("Failed to open %1 for writing: %2")
99 .arg(outputPathName, file.errorString());
100 return false;
101 }
102
103 push(additionalContext);
104 QTextStream textStream(&file);
105 NoEscapeStream noEscapeStream(&textStream);
106 Grantlee::Template tmplate = engine.loadByName(templateName);
107 if (!tmplate) {
108 qCCritical(lc).noquote() << tr("Failed to fetch template %1").arg(outputPathName);
109 pop();
110 return false;
111 }
112 tmplate->render(&noEscapeStream, &context);
113 if (tmplate->error()) {
114 qCCritical(lc).noquote() << tr("Failed to render %1: %2").arg(outputPathName, tmplate->errorString());
115 pop();
116 return false;
117 }
118 pop();
119 return true;
120}

122QString Renderer::sanitise(const QString &key)
123{
124 QString newKey = key;
125 int pos;
126 while ((pos = newKey.indexOf(QLatin1Char('.'))) >= 0) {
127 newKey = newKey.mid(0, pos) + newKey.mid(pos+1,1).toUpper() + newKey.mid(pos+2);
128 }
129 newKey.replace(QRegularExpression{QSL("[^a-zA-Z0-9_]")}, QSL("_"));
130 newKey.replace(QRegularExpression{QSL("^[0-9_]")}, QLatin1String());
131 return newKey;
132}

144QVariantHash Renderer::sanitise(const QVariantHash &hash)
145{
146 QVariantHash newHash;
147 for (auto iter = hash.begin(); iter != hash.end(); ++iter) {
148 const QString saneKey = sanitise(iter.key());
149 QVariant saneValue = iter.value();
150 if (iter->type() == QVariant::Hash) {
151 saneValue = sanitise(iter->toHash());
152 } else if (iter->type() == QVariant::Map) {
153 saneValue = sanitise(iter->toMap());
154 }
155 newHash.insert(saneKey, saneValue);
156 }
157 Q_ASSERT(hash.size() == newHash.size());
158 return newHash;
159}
