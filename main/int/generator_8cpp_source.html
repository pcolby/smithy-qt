23int Generator::expectedFileCount() const
24{
25 const QHash<smithy::ShapeId, smithy::Shape> services = model->shapes(smithy::Shape::Type::Service);
26 const int operations = std::accumulate(services.constBegin(), services.constEnd(), 0,
27 [](const int a, const smithy::Shape &shape) { return a + shape.operations().size();
28 });
29 int expected = 0;
30 const QStringList templates = renderer->templatesNames();
31 for (const QString &name: templates) {
32 if (name.contains(QSL("ServiceName"), Qt::CaseInsensitive)||
33 name.contains(QSL("Service_Name"), Qt::CaseInsensitive)) {
34 if (name.contains(QSL("OperationName"), Qt::CaseInsensitive)||
35 name.contains(QSL("Operation_Name"), Qt::CaseInsensitive)) {
36 expected += operations;
37 } else {
38 expected += services.size();
39 }
40 } else {
41 expected += 1;
42 }
43 }
44 return expected;
45}

48{

51 qCDebug(lc).noquote() << tr("Generating code for service %1").arg(service.id().toString());

53 if (sdkId.isEmpty()) {
54 qCCritical(lc).noquote() << tr("Service %1 has no SDK ID").arg(service.id().toString());
55 return -1;

58 qCDebug(lc).noquote() << tr("SDK ID: %1").arg(sdkId);

60 }
61 Q_UNUSED(outputDir);
62 Q_UNUSED(clobberMode);
63 Q_UNIMPLEMENTED();
64 return -1; /// \todo
