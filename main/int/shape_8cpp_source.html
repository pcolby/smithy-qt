34
35Shape::Shape(const QJsonObject &ast, const ShapeId &id) : d_ptr(new ShapePrivate(this))
36{
37 Q_D(Shape);
38 d->ast = ast;
39 d->error = Error::NoError;
40 d->id = id;
41 d->type = ShapePrivate::getType(ast);
42 if (d->type == Type::Undefined) {
43 d->error = Error::UndefinedShapeType;
44 }
45
46 // Warn on any unsupported properties.
47 const QStringList supportedProperties = ShapePrivate::supportedProperties(d->type);
48 const QStringList keys = ast.keys();
49 for (const QString &key: keys) {
50 if (!supportedProperties.contains(key)) {
51 qCWarning(d->lc).noquote() << tr("Ignoring unsupported Shape property: %1").arg(key);
52 }
53 }
54
55 // Error on any missing required properties.
56 const QStringList requiredProperties = ShapePrivate::requiredProperties(d->type);
57 for (const QString &key: requiredProperties) {
58 if (!keys.contains(key)) {
59 qCritical(d->lc).noquote() << tr("Missing required Shape property: %1").arg(key);
60 d->error = Error::MissingRequiredProperty;
61 }
62 }
63
64 // Validate all present properties.
65 for (auto iter = ast.constBegin(); iter != ast.constEnd(); ++iter) {
66 if (!supportedProperties.contains(iter.key())) {
67 continue;
68 }
69 if (!ShapePrivate::validateProperty(iter.key(), iter.value())) {
70 if (d->error == Error::NoError) {
71 d->error = Error::InvalidPropertyValue;
72 }
73 }
74 }
75}

304Shape::Type ShapePrivate::getType(const QJsonObject &ast)
305{
306 const QJsonValue value = ast.value(QLatin1String("type"));
307 if (value.isUndefined()) {
308 qCCritical(lc).noquote() << tr("Shape has no type property");
309 return Shape::Type::Undefined;
310 }
311 if (value.type() != QJsonValue::String) {
312 qCCritical(lc).noquote() << tr("Shape type property is not a JSON string:") << value;
313 return Shape::Type::Undefined;
314 }
315 return getType(value.toString());
316}

318Shape::Type ShapePrivate::getType(const QString &type)
319{
320 // Simple Types
321 if (type == QLatin1String("blob")) return Shape::Type::Blob;
322 if (type == QLatin1String("boolean")) return Shape::Type::Boolean;
323 if (type == QLatin1String("string")) return Shape::Type::String;
324 if (type == QLatin1String("enum")) return Shape::Type::Enum;
325 if (type == QLatin1String("byte")) return Shape::Type::Byte;
326 if (type == QLatin1String("short")) return Shape::Type::Short;
327 if (type == QLatin1String("integer")) return Shape::Type::Integer;
328 if (type == QLatin1String("intEnum")) return Shape::Type::IntEnum;
329 if (type == QLatin1String("long")) return Shape::Type::Long;
330 if (type == QLatin1String("float")) return Shape::Type::Float;
331 if (type == QLatin1String("double")) return Shape::Type::Double;
332 if (type == QLatin1String("bigInteger")) return Shape::Type::BigInteger;
333 if (type == QLatin1String("bigDecimal")) return Shape::Type::BigDecimal;
334 if (type == QLatin1String("timestamp")) return Shape::Type::Timestamp;
335 if (type == QLatin1String("document")) return Shape::Type::Document;

337 // Aggregate Types
338 if (type == QLatin1String("list")) return Shape::Type::List;
339 if (type == QLatin1String("set")) return Shape::Type::Set;
340 if (type == QLatin1String("map")) return Shape::Type::Map;

375Shape::ShapeReference ShapePrivate::getShapeRef(const QJsonObject &ast, const QString &name)
376{
377 const auto iter = ast.constFind(name);
378 if (iter == ast.constEnd()) {
379 return Shape::ShapeReference{};
380 }
381 const QJsonObject shapeRefObj = iter->toObject();
382 return Shape::ShapeReference{ shapeRefObj.value(QLatin1String("target")).toString() };
383}
384
385Shape::ShapeReferences ShapePrivate::getShapeRefs(const QJsonObject &ast, const QString &name)
386{
387 const auto iter = ast.constFind(name);
388 if (iter == ast.constEnd()) {
389 return Shape::ShapeReferences{};
390 }
391 const QJsonArray shapeRefsArray = iter->toArray();
392 Shape::ShapeReferences shapeRefs;
393 for (const QJsonValue &shapeRef: shapeRefsArray) {
394 shapeRefs.insert(Shape::ShapeReference{
395 shapeRef.toObject().value(QLatin1String("target")).toString()
396 });
397 }
398 return shapeRefs;
399}

603bool ShapePrivate::validateIdentifier(const QString &id)
604{
605 // Lazily relying on the fact that ShapeId will validate the idenitifer regex pattern for now.
606 const ShapeId shapeId(id);
607 return ((shapeId.isValid()) && (!shapeId.hasNameSpace()) && (!shapeId.hasMemberName()));
608}

610bool ShapePrivate::validateProperty(const QString &name, const QJsonValue &value)
611{
612 // Type string
613 if (name == QLatin1String("type")) {
614 if (!value.isString()) {
615 qCCritical(lc).noquote() << tr("%1 property is not a JSON string").arg(name);
616 return false;
617 }
618 return (getType(value.toString()) != Shape::Type::Undefined);
619 }
620
621 // TraitsMap
622 else if (name == QLatin1String("traits")) {
623 if (!value.isObject()) {
624 qCCritical(lc).noquote() << tr("%1 property is not a JSON object").arg(name);
625 return false;
626 }
627 const QJsonObject traits = value.toObject();
628 for (auto iter = traits.constBegin(); iter != traits.constEnd(); ++iter) {
629 const ShapeId id(iter.key());
630 if (!id.isValid()) {
631 qCCritical(lc).noquote() << tr("%1 property has trait with invalid shape ID %2")
632 .arg(name, iter.key());
633 return false;
634 }
635 if (id.isRelativeShapeId()) {
636 qCCritical(lc).noquote() << tr("%1 property has trait with relative shape ID %2")
637 .arg(name, iter.key());
638 return false;
639 }
640 // It doesn't matter (at this point) what type iter->value() is.
641 }
642 }
643
644 // Member (aka ShapeReference plus optional TraitsMap)
645 else if ((name == QLatin1String("member")) ||
646 (name == QLatin1String("key")) ||
647 (name == QLatin1String("value")))
648 {
649 if (!value.isObject()) {
650 qCCritical(lc).noquote() << tr("%1 property is not a JSON object").arg(name);
651 return false;
652 }
653 if (!validateProperty(QStringLiteral("ShapeReference"), value)) {
654 qCCritical(lc).noquote() << tr("%1 property is not valid ShapeReference").arg(name);
655 return false;
656 }
657 const QJsonObject member = value.toObject();
658 const auto traits = member.constFind(QLatin1String("traits")); // Optional.
659 if ((traits != member.constEnd()) && (!validateProperty(traits.key(), traits.value()))) {
660 qCCritical(lc).noquote() << tr("%1 property has invalid traits property").arg(name);
661 return false;
662 }
663 }
664
665 // Members (ie Set of Member instances)
666 else if (name == QLatin1String("members")) {
667 if (!value.isObject()) {
668 qCCritical(lc).noquote() << tr("%1 property is not a JSON object").arg(name);
669 return false;
670 }
671 QStringList names;
672 const QJsonObject members = value.toObject();
673 for (auto iter = members.constBegin(); iter != members.constEnd(); ++iter) {
674 if (!validateIdentifier(iter.key())) {
675 qCCritical(lc).noquote() << tr("%1 property has invalid member name %2")
676 .arg(name, iter.key());
677 return false;
678 }
679 if (names.contains(iter.key().toLower())) {
680 qCCritical(lc).noquote() << tr("%1 property has non-unique member name %2")
681 .arg(name, iter.key().toLower());
682 return false;
683 }
684 names.append(iter.key().toLower());
685 if (!validateProperty(QLatin1String("member"), iter.value())) {
686 qCCritical(lc).noquote() << tr("%1 property has invalid value for %2 property")
687 .arg(name, iter.key());
688 return false;
689 }
690 }
691 }
692
693 // Plain string
694 else if (name == QLatin1String("version")) {
695 if (!value.isString()) {
696 qCCritical(lc).noquote() << tr("%1 property is not a JSON string").arg(name);
697 return false;
698 }
699 // No further validation required; the string is free-form.
700 }
701
702 // ShapeReferences
703 else if ((name == QLatin1String("operations")) ||
704 (name == QLatin1String("collectionOperations")) ||
705 (name == QLatin1String("resources")) ||
706 (name == QLatin1String("errors")) ||
707 (name == QLatin1String("mixins"))) {
708 if (!value.isArray()) {
709 qCCritical(lc).noquote() << tr("%1 property is not a JSON array").arg(name);
710 return false;
711 }
712 const QJsonArray shapeRefs = value.toArray();
713 for (const QJsonValue &shapeRef: shapeRefs) {
714 if (!validateProperty(QLatin1String("ShapeReference"), shapeRef)) {
715 qCCritical(lc).noquote() << tr("%1 property has invalid entry").arg(name);
716 }
717 }
718 }
719
720 // ShapeIdStringMap
721 else if (name == QLatin1String("rename")) {
722 if (!value.isObject()) {
723 qCCritical(lc).noquote() << tr("%1 property is not a JSON object").arg(name);
724 return false;
725 }
726 const QJsonObject rename = value.toObject();
727 for (auto iter = rename.constBegin(); iter != rename.constEnd(); ++iter) {
728 const ShapeId shapeId(iter.key());
729 if (!shapeId.isValid()) {
730 qCCritical(lc).noquote() << tr("%1 property has invalid shape ID %2")
731 .arg(name, iter.key());
732 return false;
733 }
734 if (!iter.value().isString()) {
735 qCCritical(lc).noquote() << tr("%1 property has shape ID %2 with invalid value")
736 .arg(name, iter.key());
737 return false;
738 }
739 const QString indentifier = iter.value().toString();
740 if (validateIdentifier(indentifier)) {
741 qCCritical(lc).noquote() << tr("%1 property has shape ID %2 with invalid identifier %3")
742 .arg(name, iter.key(), indentifier);
743 return false;
744 }
745 }
746 }
747
748 // ShapeReference
749 else if ((name == QLatin1String("ShapeReference")) ||
750 (name == QLatin1String("create")) ||
751 (name == QLatin1String("put")) ||
752 (name == QLatin1String("read")) ||
753 (name == QLatin1String("update")) ||
754 (name == QLatin1String("delete")) ||
755 (name == QLatin1String("list")) ||
756 (name == QLatin1String("input")) ||
757 (name == QLatin1String("output"))) {
758 if (!value.isObject()) {
759 qCCritical(lc).noquote() << tr("%1 property is not a JSON object").arg(name);
760 return false;
761 }
762 const QJsonObject shapeRef = value.toObject();
763 const auto target = shapeRef.constFind(QLatin1String("target")); // Required.
764 if (target == shapeRef.constEnd()) {
765 qCCritical(lc).noquote() << tr("%1 property has no target property").arg(name);
766 return false;
767 }
768 if (!target.value().isString()) {
769 qCCritical(lc).noquote() << tr("%1 property has target value that not a JSON string")
770 .arg(name);
771 return false;
772 }
773 const QString shapeIdString = target.value().toString();
774 const ShapeId shapeId(shapeIdString);
775 if (!shapeId.isValid()) {
776 qCCritical(lc).noquote() << tr("%1 property has target with invalid shape ID %2")
777 .arg(name, shapeIdString);
778 return false;
779 }
780 if (shapeId.isRelativeShapeId()) {
781 qCCritical(lc).noquote() << tr("%1 property has target with relative shape ID %2")
