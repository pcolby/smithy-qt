37void configureLogging(const QCommandLineParser &parser)
38{
39 // Start with the Qt default message pattern (see qtbase:::qlogging.cpp:defaultPattern)
40 QString messagePattern = QStringLiteral("%{if-category}%{category}: %{endif}%{message}");
41
42 if (parser.isSet(QStringLiteral("debug"))) {
43 #ifdef QT_MESSAGELOGCONTEXT
44 // %{file}, %{line} and %{function} are only available when QT_MESSAGELOGCONTEXT is set.
45 messagePattern.prepend(QStringLiteral("%{function} "));
46 #endif
47 messagePattern.prepend(QStringLiteral("%{time process} %{type} "));
48 QLoggingCategory::setFilterRules(QStringLiteral("smithy.*.debug=true"));
49 }
50
51 const QString color = parser.value(QStringLiteral("color"));
52 if ((color == QStringLiteral("yes")) || (color == QStringLiteral("auto") && haveConsole())) {
53 messagePattern.prepend(QStringLiteral(
54 "%{if-debug}\x1b[37m%{endif}" // White
55 "%{if-info}\x1b[32m%{endif}" // Green
56 "%{if-warning}\x1b[35m%{endif}" // Magenta
57 "%{if-critical}\x1b[31m%{endif}" // Red
58 "%{if-fatal}\x1b[31;1m%{endif}")); // Red and bold
59 messagePattern.append(QStringLiteral("\x1b[0m")); // Reset.
60 }
61
62 qSetMessagePattern(messagePattern);
63}

89bool requireOptions(const QStringList &requiredOtions, const QCommandLineParser &parser)
90{
91 QStringList missingOptions;
92 for (const QString &requiredOption: requiredOtions) {
93 if (!parser.isSet(requiredOption)) {
94 missingOptions.append(requiredOption);
95 }
96 }
97 if (!missingOptions.empty()) {
98 qCCritical(lc).noquote() << QCoreApplication::translate("requireOptions",
99 "Missing required option(s): %1").arg(missingOptions.join(QLatin1Char(' ')));
100 return false;
101 }
102 return true;
103}

115bool requireDirs(const QCommandLineParser &parser, const QString &option, const QDir::Filters &rw)
116{
117 bool success = true;
118 const QStringList dirs = parser.values(option);
119 for (const QString &dir: dirs) {
120 const QFileInfo info(dir);
121 const QString label = option.at(0).toUpper() + option.mid(1);
122 if (!info.exists()) {
123 qCCritical(lc).noquote() << QCoreApplication::translate("requireDirs",
124 "%1 directory does not exist: %2").arg(label, dir);
125 success = false;
126 } else if (!info.isDir()) {
127 qCCritical(lc).noquote() << QCoreApplication::translate("requireDirs",
128 "%1 directory is not a directory: %2").arg(label, dir);
129 success = false;
130 } else {
131 if ((rw.testFlag(QDir::Readable)) && (!info.isReadable())) {
132 qCCritical(lc).noquote() << QCoreApplication::translate("requireDirs",
133 "%1 directory is not readable: %2").arg(label, dir);
134 success = false;
135 }
136 if ((rw.testFlag(QDir::Writable)) && (!info.isWritable())) {
137 qCCritical(lc).noquote() << QCoreApplication::translate("requireDirs",
138 "%1 directory is not writable: %2").arg(label, dir);
139 success = false;
140 }
141 }
142 }
143 return success;
144}

146inline bool checkRequiredDirs(const QCommandLineParser &parser)
147{
148 bool success = true;
149 const QMap<QString,QDir::Filters> options{
150 { QLatin1String("models"), QDir::Readable },
151 { QLatin1String("templates"), QDir::Readable },
152 { QLatin1String("output"), QDir::Writable },
153 };
154 for (auto iter = options.constBegin(); iter != options.constEnd(); ++iter) {
155 if (!requireDirs(parser, iter.key(), iter.value())) {
156 success = false;
157 }
158 }
159 return success;
160}

162int loadModels(const QString &dir, smithy::Model &model)
163{
164 int count = 0;
165 qCInfo(lc).noquote()
166 << QCoreApplication::translate("loadModels", "Loading Smithy models from %1").arg(dir);
167 QDirIterator iter{dir, {QStringLiteral("*.json")}, QDir::Files, QDirIterator::Subdirectories};
168 while (iter.hasNext()) {
169 QFile file{iter.next()};
170 qCDebug(lc).noquote() << QCoreApplication::translate("loadModels", "Loading Smithy JSON: %1")
171 .arg(file.fileName());
172 file.open(QFile::ReadOnly);
173 QJsonParseError error{};
174 QJsonDocument json = QJsonDocument::fromJson(file.readAll(), &error);
175 if (error.error != QJsonParseError::NoError) {
176 qCCritical(lc).noquote() << QCoreApplication::translate("loadModels",
177 "Failed to parse JSON file: %1").arg(file.fileName());
178 return -count;
179 }
180 if (!json.isObject()) {
181 qCCritical(lc).noquote() << QCoreApplication::translate("loadModels",
182 "File is not a JSON object: %1").arg(file.fileName());
183 return -count;
184 }
185 if (!model.insert(json.object())) {
186 qCCritical(lc).noquote() << QCoreApplication::translate("loadModels",
187 "Failed to parse Smithy JSON AST: %1").arg(file.fileName());
188 return -count;
189 }
190 ++count;
191 }
192 qCDebug(lc).noquote() << QCoreApplication::translate("loadModels", "Loaded %n model(s) from %1",
193 nullptr, count).arg(dir);
194 if (count == 0) {
195 qCCritical(lc).noquote() << QCoreApplication::translate("loadModels",
196 "Failed to find any JSON AST models in %1").arg(dir);
197 }
198 return count;
199}

201int loadModels(const QStringList &dirs, smithy::Model &model)
202{
203 int count = 0;
204 for (const QString &dir: dirs) {
205 const int thisCount = loadModels(dir, model);
206 if (thisCount <= 0) {
207 return thisCount - count;
208 } else {
209 count += thisCount;
210 }
211 }
212 qCDebug(lc).noquote() << QCoreApplication::translate("loadModels", "Loaded %n model(s) in total",
213 nullptr, count);
214 return count;
215}

217int main(int argc, char *argv[])
218{
219 // Setup the core application.
220 QCoreApplication app(argc, argv);
221 app.setApplicationName(QStringLiteral(PROJECT_NAME));
222 #ifdef PROJECT_PRE_RELEASE
223 app.setApplicationVersion(QStringLiteral(PROJECT_VERSION "-" PROJECT_PRE_RELEASE));
224 #else
225 app.setApplicationVersion(QStringLiteral(PROJECT_VERSION));
226 #endif
227
228 // Parse the command line options.
229 QCommandLineParser parser;
230 parseCommandLineOptions(app, parser);
231 configureLogging(parser);
232 if (!checkRequiredOptions(parser)) return 1;
233 if (!checkRequiredDirs(parser)) return 2;
234
235 // Load the Smithy model files.
236 smithy::Model model;
237 const int modelsCount = loadModels(parser.values(QStringLiteral("models")), model);
238 if (modelsCount <= 0) return 3; // loadModels() will have already logged the (criticial) error.
239 if ((!model.finish()) || (!model.isValid())) {
240 qCCritical(lc).noquote() << QCoreApplication::translate("main",
241 "Failed to merge Smithy model files into a valid Smithy model");
242 return 4;
243 }
244
245 // Setup the renderer.
246 Renderer renderer;
247 if (!renderer.loadTemplates(parser.value(QStringLiteral("templates")))) {
248 return 5;
249 }
250
251 // Setup the generator.
252 const QString outputDir = QDir(parser.value(QStringLiteral("output"))).absolutePath();
253 Generator generator(&model, &renderer);
254 qCWarning(lc).noquote() << QCoreApplication::translate("main", "About to generate approximately"
255 " %n file/s in %2", nullptr, generator.expectedFileCount()).arg(outputDir);
256 if (!parser.isSet(QStringLiteral("force"))) {
257 qCInfo(lc).noquote() << QCoreApplication::translate("main", "Press Enter to contine");
258 QTextStream stream(stdin);
259 stream.readLine();
260 }
261 if (!generator.generate(outputDir, parser.isSet(QStringLiteral("force"))
262 ? Generator::ClobberMode::Overwrite : Generator::ClobberMode::Prompt)) {
263 return 6;
264 }
265 qCInfo(lc).noquote() << QCoreApplication::translate("main",
266 "Rendered %n file/s in %2", nullptr, -1).arg(outputDir);
267 return 0;
268}
